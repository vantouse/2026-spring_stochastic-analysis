def distribution_error(
    p: np.ndarray,
    p_true: float = 1.
):
    return np.sum((p - p_true) ** 2)

def pearson_r(x: np.ndarray, y: np.ndarray):
    return np.sum((x - x.mean()) * (y - y.mean())) / np.sqrt(np.sum((x - x.mean()) ** 2) * np.sum((y - y.mean()) ** 2))
